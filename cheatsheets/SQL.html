SQL Cheat Sheet

SQL Functions

Function Description Example
SELECTRetrieves data from one or more tables.SELECT column1, column2 FROM table_name WHERE condition;
DISTINCTRemoves duplicate records from the result.SELECT DISTINCT column1 FROM table_name;
WHEREFilters records based on conditions.SELECT * FROM table_name WHERE column1 = 'value';
ORDER BYSorts the result in ascending or descending order.SELECT * FROM table_name ORDER BY column1 ASC;
GROUP BYGroups rows sharing a property.SELECT column1, COUNT(*) FROM table_name GROUP BY column1;
HAVINGFilters groups defined by GROUP BY.SELECT column1, COUNT(*) FROM table_name GROUP BY column1 HAVING COUNT(*) > 1;
LIMITLimits the number of rows returned.SELECT * FROM table_name LIMIT 10;
INSERT INTOInserts new rows into a table.INSERT INTO table_name (column1, column2) VALUES ('value1', 'value2');
UPDATEUpdates existing rows in a table.UPDATE table_name SET column1 = 'value' WHERE condition;
DELETEDeletes rows from a table.DELETE FROM table_name WHERE condition;
CREATE TABLECreates a new table.CREATE TABLE table_name (column1 datatype, column2 datatype);
ALTER TABLEModifies an existing table (e.g., add/remove columns).ALTER TABLE table_name ADD column_name datatype;
DROP TABLEDeletes a table completely.DROP TABLE table_name;
INNER JOINReturns rows with matching values in both tables.SELECT table1.column1, table2.column2 FROM table1 INNER JOIN table2 ON table1.common_column = table2.common_column;
LEFT JOINReturns all rows from the left table, even if no match is found.SELECT * FROM table1 LEFT JOIN table2 ON table1.common_column = table2.common_column;
RIGHT JOINReturns all rows from the right table, even if no match is found.SELECT * FROM table1 RIGHT JOIN table2 ON table1.common_column = table2.common_column;
FULL JOINReturns rows when there's a match in either table.SELECT * FROM table1 FULL JOIN table2 ON table1.common_column = table2.common_column;
UNIONCombines results of two SELECT statements.SELECT column1 FROM table1 UNION SELECT column1 FROM table2;
CASEAdds conditional logic to queries.SELECT column1, CASE WHEN condition1 THEN 'Result1' ELSE 'Default' END AS alias FROM table_name;
SUBQUERYEmbeds a query within another query.SELECT column1 FROM table_name WHERE column2 = (SELECT column2 FROM another_table);
CREATE INDEXCreates an index on a table.CREATE INDEX index_name ON table_name (column_name);
CREATE VIEWCreates a virtual table based on a query.CREATE VIEW view_name AS SELECT column1 FROM table_name WHERE condition;
BEGIN TRANSACTIONStarts a new transaction.BEGIN TRANSACTION;
COMMITSaves changes made during the transaction.COMMIT;
ROLLBACKReverts changes made during the transaction.ROLLBACK;
COUNT()Counts the number of rows.SELECT COUNT(*) FROM table_name;
SUM()Adds up numeric values.SELECT SUM(column1) FROM table_name;
AVG()Calculates the average of numeric values.SELECT AVG(column1) FROM table_name;
MIN()Finds the smallest value.SELECT MIN(column1) FROM table_name;
MAX()Finds the largest value.SELECT MAX(column1) FROM table_name;
CONCAT()Combines strings.SELECT CONCAT(column1, ' ', column2) AS FullName FROM table_name;
LENGTH()Returns the length of a string.SELECT LENGTH(column1) FROM table_name;
UPPER()Converts text to uppercase.SELECT UPPER(column1) FROM table_name;
LOWER()Converts text to lowercase.SELECT LOWER(column1) FROM table_name;
SUBSTRING()Extracts a part of a string.SELECT SUBSTRING(column1, 1, 5) FROM table_name;
TRIM()Removes leading and trailing spaces.SELECT TRIM(column1) FROM table_name;
NOW()Returns the current date and time.SELECT NOW();
DATE()Extracts the date part.SELECT DATE(column1) FROM table_name;
DATEDIFF()Calculates the difference between two dates.SELECT DATEDIFF(NOW(), column1) AS DaysPassed FROM table_name;
DATE_ADD()Adds a time interval to a date.SELECT DATE_ADD(column1, INTERVAL 7 DAY) FROM table_name;
DATE_FORMAT()Formats date values.SELECT DATE_FORMAT(column1, '%Y-%m-%d') FROM table_name;
